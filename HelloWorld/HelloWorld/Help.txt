public void StartMultiply()
        {
            var firstNumber = GetInputNumber("first");
            var secondNumber = GetInputNumber("second");
            var res = firstNumber * secondNumber;
            Console.WriteLine(firstNumber + " * " + secondNumber + " = " + res);

            Console.ReadLine();
        }
